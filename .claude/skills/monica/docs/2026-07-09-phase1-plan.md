```python
#!/usr/bin/env python3
"""monica — CLI for the Monica CRM v4 REST API. Stdlib only.

Config: env vars MONICA_API_URL / MONICA_API_TOKEN, else ~/.config/monica-api/.env
Output: JSON on stdout. Errors on stderr; exit 1 = API error, 2 = not configured.
"""
import argparse, json, os, ssl, sys, urllib.error, urllib.parse, urllib.request
from pathlib import Path

CONFIG_FILE = Path.home() / ".config" / "monica-api" / ".env"
SETUP_MSG = """monica: not configured.
Set MONICA_API_URL and MONICA_API_TOKEN (env vars, or KEY=VALUE lines in
~/.config/monica-api/.env, chmod 600). Get a token in Monica: Settings -> API.
See .env.example in this skill directory."""

def load_config(config_file=CONFIG_FILE):
    cfg = {}
    if Path(config_file).exists():
        for line in Path(config_file).read_text().splitlines():
            line = line.strip()
            if line and not line.startswith("#") and "=" in line:
                k, v = line.split("=", 1)
                cfg[k.strip()] = v.strip()
    for k in ("MONICA_API_URL", "MONICA_API_TOKEN", "MONICA_API_CA_BUNDLE"):
        if os.environ.get(k):
            cfg[k] = os.environ[k]
    return cfg

def fail(msg, code=1):
    sys.stderr.write(msg.rstrip() + "\n")
    sys.exit(code)

def request(cfg, method, path, params=None, body=None):
    url = cfg["MONICA_API_URL"].rstrip("/") + "/" + path.lstrip("/")
    if params:
        url += "?" + urllib.parse.urlencode(params)
    data = json.dumps(body).encode() if body is not None else None
    req = urllib.request.Request(url, data=data, method=method, headers={
        "Authorization": "Bearer " + cfg["MONICA_API_TOKEN"],
        "Accept": "application/json",
        "Content-Type": "application/json",
    })
    # TLS verification always on; self-signed instances supply their CA via MONICA_API_CA_BUNDLE
    ctx = ssl.create_default_context(cafile=cfg["MONICA_API_CA_BUNDLE"]) if cfg.get("MONICA_API_CA_BUNDLE") else None
    try:
        with urllib.request.urlopen(req, context=ctx) as r:
            raw = r.read()
            return json.loads(raw) if raw else {}
    except urllib.error.HTTPError as e:
        try:
            err = json.loads(e.read())
        except Exception:
            err = {}
        detail = err.get("error", {}) if isinstance(err.get("error"), dict) else {}
        code = detail.get("error_code")
        msg = detail.get("message") or err.get("message") or str(e.reason)
        if isinstance(err.get("errors"), dict):        # 422 validation detail
            msg += " " + json.dumps(err["errors"], ensure_ascii=False)
        if e.code == 429:
            msg += " (rate limited — wait a minute and retry)"
        fail("HTTP %d%s: %s" % (e.code, " [monica error %s]" % code if code else "", msg))
    except urllib.error.URLError as e:
        fail("connection error: %s" % e.reason)

def paginate(cfg, path, params=None):
    """Follow meta.last_page; return concatenated data list."""
    params = dict(params or {})
    items, page = [], 1
    while True:
        params["page"] = page
        resp = request(cfg, "GET", path, params)
        items.extend(resp.get("data", []))
        meta = resp.get("meta", {})
        if page >= meta.get("last_page", 1):
            return items
        page += 1

def out(obj):
    print(json.dumps(obj, ensure_ascii=False))

def cmd_whoami(cfg, args):
    out(request(cfg, "GET", "me")["data"])

def main(argv=None):
    p = argparse.ArgumentParser(prog="monica", description=__doc__)
    sub = p.add_subparsers(dest="resource", required=True)
    sub.add_parser("whoami", help="current user + me_contact")
    args = p.parse_args(argv)
    cfg = load_config()
    if not cfg.get("MONICA_API_URL") or not cfg.get("MONICA_API_TOKEN"):
        fail(SETUP_MSG, 2)
    HANDLERS[args.resource](cfg, args)

HANDLERS = {"whoami": cmd_whoami}

if __name__ == "__main__":
    main()
```

```python
# Full field set required by PUT /contacts/{id} (UpdateContact::rules() in Monica v4):
# first_name(req), middle_name, last_name, nickname, gender_id, description,
# is_birthdate_known(req) + birthdate_day/month/year + birthdate_is_age_based + birthdate_age,
# is_deceased(req-ish), is_deceased_date_known(req) + deceased_date_day/month/year.
# Omitted fields are WIPED — hence GET-merge-PUT.

def expand_birthdate(payload):
    """Accept birthdate: 'YYYY-MM-DD' shorthand and expand to Monica's composite params."""
    p = dict(payload)
    bd = p.pop("birthdate", None)
    if bd:
        y, m, d = (int(x) for x in bd.split("-"))
        p.update(is_birthdate_known=True, birthdate_day=d, birthdate_month=m,
                 birthdate_year=y, birthdate_is_age_based=False)
    p.setdefault("is_birthdate_known", False)
    p.setdefault("is_deceased", False)
    p.setdefault("is_deceased_date_known", False)
    return p

def _date_parts(dateinfo):
    date = (dateinfo or {}).get("date")
    if not date:
        return None
    d = date[:10].split("-")
    return {"year": int(d[0]), "month": int(d[1]), "day": int(d[2])}

def contact_to_update_payload(c):
    """Map GET /contacts/{id} resource back to the full PUT payload."""
    info = c.get("information", {})
    bd = _date_parts(info.get("dates", {}).get("birthdate"))
    dd = _date_parts(info.get("dates", {}).get("deceased_date"))
    p = {
        "first_name": c.get("first_name"),
        "middle_name": c.get("middle_name"),
        "last_name": c.get("last_name"),
        "nickname": c.get("nickname"),
        "gender_id": c.get("gender_id") or info.get("gender_id"),
        "description": c.get("description"),
        "is_birthdate_known": bool(bd),
        "is_deceased": bool(c.get("is_dead")),
        "is_deceased_date_known": bool(dd),
    }
    if bd:
        p.update(birthdate_day=bd["day"], birthdate_month=bd["month"],
                 birthdate_year=bd["year"],
                 birthdate_is_age_based=bool(info["dates"]["birthdate"].get("is_age_based")))
    if dd:
        p.update(deceased_date_day=dd["day"], deceased_date_month=dd["month"],
                 deceased_date_year=dd["year"])
    return p
```

CACHE_DIR = Path.home() / ".cache" / "monica-skill"
REF_PATHS = {"genders": "genders", "fieldtypes": "contactfieldtypes", "reltypes": "relationshiptypes"}

def ref_lookup(cfg, kind, refresh=False):
    host = urllib.parse.urlparse(cfg["MONICA_API_URL"]).netloc.replace(":", "_")
    cache = CACHE_DIR / ("%s-%s.json" % (host, kind))
    if cache.exists() and not refresh:
        return json.loads(cache.read_text())
    items = paginate(cfg, REF_PATHS[kind])
    CACHE_DIR.mkdir(parents=True, exist_ok=True)
    cache.write_text(json.dumps(items))
    return items

def cmd_ref(cfg, args):
    out(ref_lookup(cfg, args.kind, refresh=args.refresh))
```

def cmd_notes(cfg, args):
    if args.action == "add":
        out(request(cfg, "POST", "notes", body={"contact_id": args.contact, "body": args.text})["data"])
    elif args.action == "list":
        out(request(cfg, "GET", "contacts/%d/notes" % args.contact)["data"])
    elif args.action == "delete":
        out(request(cfg, "DELETE", "notes/%d" % args.note_id))

def cmd_tags(cfg, args):
    if args.action == "list":
        out(paginate(cfg, "tags"))
    elif args.action == "set":
        out(request(cfg, "POST", "contacts/%d/setTags" % args.contact,
                    body={"tags": args.tags.split(",")})["data"])
    elif args.action == "unset":
        out(request(cfg, "POST", "contacts/%d/unsetTag" % args.contact,
                    body={"tag_id": find_tag_id(cfg, args.tag)})["data"])

def cmd_reminders(cfg, args):
    if args.action == "add":
        out(request(cfg, "POST", "reminders", body={
            "contact_id": args.contact, "title": args.title, "initial_date": args.date,
            "frequency_type": args.frequency, "frequency_number": 1})["data"])
    elif args.action == "list":
        out(request(cfg, "GET", "contacts/%d/reminders" % args.contact)["data"])
    elif args.action == "delete":
        out(request(cfg, "DELETE", "reminders/%d" % args.reminder_id))

def cmd_tasks(cfg, args):
    if args.action == "add":
        out(request(cfg, "POST", "tasks", body={
            "contact_id": args.contact, "title": args.title, "completed": 0})["data"])
    elif args.action == "done":
        t = request(cfg, "GET", "tasks/%d" % args.task_id)["data"]
        out(request(cfg, "PUT", "tasks/%d" % args.task_id, body={
            "contact_id": t["contact"]["id"] if t.get("contact") else None,
            "title": t["title"], "description": t.get("description"),
            "completed": 1})["data"])
    elif args.action == "list":
        out(request(cfg, "GET", "contacts/%d/tasks" % args.contact)["data"])
    elif args.action == "delete":
        out(request(cfg, "DELETE", "tasks/%d" % args.task_id))
```
